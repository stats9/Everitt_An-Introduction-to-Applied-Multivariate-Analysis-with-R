pfun2 <- function(nf){
    model <- factanal(covmat = cor_mat2, factors = nf, 
    n.obs = 1634, method = "ml")
    est <- tcrossprod(model$loadings) + diag(model$uniquenesses)
    ret <- round(cor_mat2 - est, 3)
    colnames(ret) <- rownames(ret) <- rownames(cor_mat2)
    return(ret)
}
